#Augmented Assignment operators

>>> spam = 42 
>>> spam += 1 
>>> spam 
43


Augmented assignment statement		 Equivalent assignment statement
    spam = spam + 1 			       spam += 1
    spam = spam - 1 			       spam -= 1
    spam = spam * 1 			       spam *= 1
    spam = spam / 1 			       spam /= 1
    spam = spam % 1 			       spam %= 1


>>> spam = 'Hello' 
>>> spam += ' world!' 
>>> spam 
'Hello world!'

>>> bacon = ['Zophie']
